    // Otherwise, recurse into the expression

- [ ] **Step 3: Implement the substitution**

In `crates/vexil-lang/src/substitute.rs`:

```rust
use crate::ast::TypeExpr;
use crate::ir::{ResolvedType, TypeId};
use smol_str::SmolStr;

pub fn substitute(
    expr: &TypeExpr,
    type_params: &[SmolStr],
    type_args: &[TypeExpr],
) -> ResolvedType {
    match expr {
        TypeExpr::Named(name) => {
            // Check if this name is a type parameter
            if let Some(idx) = type_params.iter().position(|p| p.as_str() == name.as_str()) {
                // Substitute with the corresponding type arg
                // For now, just resolve the arg directly
                // A full implementation would need to handle nested generics
                resolve_type_expr(&type_args[idx], /* ctx needed */)
            } else {
                // Not a type param - resolve normally
                resolve_type_expr(expr, /* ctx */)
            }
        }
        // ... handle other variants
        _ => resolve_type_expr(expr, /* ctx */)
    }
}
```

Actually, the substitution needs to happen in the IR context. It may be better to implement directly in `resolve_type_expr` in `lower.rs` where we have access to the context.

- [ ] **Step 4: Update resolve_type_expr**

In `lower.rs` in the `TypeExpr::Generic` match arm (line 1069), instead of emitting an error, implement the actual substitution:

```rust
TypeExpr::Generic(name, arg) => {
    let alias_id = ctx.registry.lookup(name.as_str());
    match alias_id {
        Some(id) => {
            if let Some(TypeDef::GenericAlias(alias_def)) = ctx.registry.get(id) {
                // Perform type substitution
                let resolved = substitute_type_expr(
                    &alias_def.target_type,
                    &alias_def.type_params,
                    &[arg.node.clone()],
                    ctx,
                );
                resolved
            } else {
                ctx.emit(span, ErrorClass::AliasTargetNotFound, ...);
                ResolvedType::Named(ir::types::POISON_TYPE_ID)
            }
        }
        None => {
            ctx.emit(span, ErrorClass::UnresolvedType, ...);
            ResolvedType::Named(ir::types::POISON_TYPE_ID)
        }
    }
}
```
